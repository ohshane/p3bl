import { format, differenceInDays } from 'date-fns'
import {
  FileText,
  Bot,
  Users,
  Calendar,
  ListChecks,
  UserCheck,
  CheckCircle,
  AlertCircle,
} from 'lucide-react'
import { useCreatorStore } from '@/stores/creatorStore'
import { Badge } from '@/components/ui/badge'
import { cn } from '@/lib/utils'
 
export function ReviewAndDeploy() {
  const { wizardState, aiPersonas, validateWizard } = useCreatorStore()
  const {
    mode,
    basicInfo,
    selectedAIPersonaIds,
    participantParams,
    timeline,
    sessions,
    invitedExpertIds,
    isValid,
    validationErrors,
  } = wizardState
 
  // Run validation
  const isValidProject = validateWizard()
 
  const duration = timeline.startDate && timeline.endDate
    ? differenceInDays(new Date(timeline.endDate), new Date(timeline.startDate))
    : 0
 
  const selectedPersonas = aiPersonas.filter(p => selectedAIPersonaIds.includes(p.id))
  const expectedTeams = Math.ceil(participantParams.totalParticipants / participantParams.teamSize)
 
  const sections = [
    {
      icon: FileText,
      title: 'Project Information',
      valid: !!basicInfo.title && !!basicInfo.drivingQuestion,
      content: (
        <div className="space-y-2">
          <div>
            <span className="text-gray-400">Title: </span>
            <span className="text-white">{basicInfo.title || <span className="text-red-400">Not set</span>}</span>
          </div>
          {basicInfo.background && (
            <div>
              <span className="text-gray-400">Background: </span>
              <span className="text-white line-clamp-2">{basicInfo.background}</span>
            </div>
          )}
          <div>
            <span className="text-gray-400">Driving Question: </span>
            <span className="text-white">{basicInfo.drivingQuestion || <span className="text-red-400">Not set</span>}</span>
          </div>
        </div>
      ),
    },
    {
      icon: Bot,
      title: 'AI Personas',
      valid: true, // Optional
      content: (
        <div>
          {selectedPersonas.length === 0 ? (
            <span className="text-gray-400">No AI personas selected</span>
          ) : (
            <div className="flex flex-wrap gap-2">
              {selectedPersonas.map(p => (
                <Badge key={p.id} variant="outline" className="border-slate-600">
                  {p.name}
                </Badge>
              ))}
            </div>
          )}
        </div>
      ),
    },
    {
      icon: Users,
      title: 'Team Configuration',
      valid: participantParams.totalParticipants > 0 && participantParams.teamSize > 0,
      content: (
        <div className="space-y-1">
          <div>
            <span className="text-gray-400">Participants: </span>
            <span className="text-white">{participantParams.totalParticipants}</span>
          </div>
          <div>
            <span className="text-gray-400">Team Size: </span>
            <span className="text-white">{participantParams.teamSize}</span>
          </div>
          <div>
            <span className="text-gray-400">Expected Teams: </span>
            <span className="text-white">{expectedTeams}</span>
          </div>
          <div>
            <span className="text-gray-400">Formation: </span>
            <span className="text-white capitalize">{participantParams.formationMode}</span>
          </div>
        </div>
      ),
    },
    {
      icon: Calendar,
      title: 'Timeline',
      valid: !!timeline.startDate && !!timeline.endDate,
      content: (
        <div className="space-y-1">
          <div>
            <span className="text-gray-400">Duration: </span>
            <span className="text-white">{duration} days</span>
          </div>
          {timeline.startDate && (
            <div>
              <span className="text-gray-400">Start: </span>
              <span className="text-white">{format(new Date(timeline.startDate), 'MMM d, yyyy')}</span>
            </div>
          )}
          {timeline.endDate && (
            <div>
              <span className="text-gray-400">End: </span>
              <span className="text-white">{format(new Date(timeline.endDate), 'MMM d, yyyy')}</span>
            </div>
          )}
        </div>
      ),
    },
    {
      icon: ListChecks,
      title: 'Sessions',
      valid: sessions.length > 0,
      content: (
        <div>
          {sessions.length === 0 ? (
            <span className="text-red-400">No sessions defined</span>
          ) : (
            <div className="space-y-1">
              <div>
                <span className="text-gray-400">Total Sessions: </span>
                <span className="text-white">{sessions.length}</span>
              </div>
              <div className="flex flex-wrap gap-1 mt-2">
                {sessions.map((s, i) => (
                  <Badge key={i} variant="outline" className="border-slate-600 text-xs">
                    {s.title}
                  </Badge>
                ))}
              </div>
            </div>
          )}
        </div>
      ),
    },
    {
      icon: UserCheck,
      title: 'Expert Reviewers',
      valid: true, // Optional
      content: (
        <div>
          {invitedExpertIds.length === 0 ? (
            <span className="text-gray-400">No experts invited</span>
          ) : (
            <span className="text-white">{invitedExpertIds.length} expert(s) invited</span>
          )}
        </div>
      ),
    },
  ]
 
  return (
    <div className="space-y-6">
      <div>
        <h2 className="text-2xl font-bold text-white mb-2">Review & Deploy</h2>
        <p className="text-gray-400">
          Review your project configuration before creating it
        </p>
      </div>
 
      {/* Validation Status */}
      <div
        className={cn(
          'p-4 rounded-lg border',
          isValidProject
            ? 'bg-green-500/10 border-green-500/30'
            : 'bg-red-500/10 border-red-500/30'
        )}
      >
        <div className="flex items-center gap-3">
          {isValidProject ? (
            <>
              <CheckCircle className="w-5 h-5 text-green-400" />
              <span className="text-green-400 font-medium">Your project is ready to be created!</span>
            </>
          ) : (
            <>
              <AlertCircle className="w-5 h-5 text-red-400" />
              <span className="text-red-400 font-medium">Please fix the following issues:</span>
            </>
          )}
        </div>
        {!isValidProject && Object.keys(validationErrors).length > 0 && (
          <ul className="mt-2 ml-8 list-disc text-sm text-red-400">
            {Object.entries(validationErrors).map(([key, error]) => (
              <li key={key}>{error}</li>
            ))}
          </ul>
        )}
      </div>
 
      {/* Summary Sections */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        {sections.map((section, index) => {
          const Icon = section.icon
 
          return (
            <div
              key={index}
              className={cn(
                'p-4 rounded-lg border',
                section.valid
                  ? 'bg-slate-800/50 border-slate-700'
                  : 'bg-red-500/5 border-red-500/30'
              )}
            >
              <div className="flex items-center gap-2 mb-3">
                <Icon className={cn('w-4 h-4', section.valid ? 'text-cyan-400' : 'text-red-400')} />
                <h3 className="font-medium text-white">{section.title}</h3>
                {section.valid ? (
                  <CheckCircle className="w-4 h-4 text-green-400 ml-auto" />
                ) : (
                  <AlertCircle className="w-4 h-4 text-red-400 ml-auto" />
                )}
              </div>
              <div className="text-sm">{section.content}</div>
            </div>
          )
        })}
      </div>
 
      {/* Deploy Info */}
      <div className="p-4 bg-slate-800/30 rounded-lg border border-slate-700">
        <h4 className="font-medium text-white mb-2">What happens next?</h4>
        <ul className="text-sm text-gray-400 space-y-1 list-disc list-inside">
          <li>Your project will be created in <strong className="text-cyan-400">Draft</strong> status</li>
          <li>A unique join code will be generated for learners</li>
          <li>You can review and edit the project before activating it</li>
          <li>Activate the project when you're ready for learners to join</li>
        </ul>
      </div>
    </div>
  )
}
